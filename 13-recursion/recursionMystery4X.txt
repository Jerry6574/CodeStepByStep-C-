int recursionMystery4X(string& s, int i, int j, char k) {
    if (i >= j) {
        return i;
    } else if (s[i] < k) {
        return recursionMystery4X(s, i + 1, j, k);
    } else if (s[j] > k) {
        return recursionMystery4X(s, i, j - 1, k);
    } else {
        int temp = s[i];
        s[i] = s[j];
        s[j] = temp;
        return recursionMystery4X(s, i + 1, j - 1, k);
    }
}
